```c
typedef struct QNode {
    int data;// 数据域
    struct QNode *next;// 指针域
} QNode;// 队列结点类型定义
```

```c
typedef struct{
    QNode* front;// 队头指针
    QNode* rear;// 队尾指针
} LinkedQueue;// 链队类型定义
```
